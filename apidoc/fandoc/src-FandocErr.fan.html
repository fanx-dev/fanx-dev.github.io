**
** FandocErr
**
@Js
const class FandocErr : Err
{

  new make(Str msg, Str file, Int line, Err? cause := null)
    : super(msg, cause)
  {
    this.file = file
    this.line = line
  }

  override Str toStr()
  {
    return "$msg [$file:$line]"
  }

  const Str file
  const Int line
}
